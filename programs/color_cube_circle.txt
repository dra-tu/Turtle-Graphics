FUN CUBE | length
	MOVE length
	ROTATE L | 90
	MOVE length
	ROTATE L | 90
	MOVE length
	ROTATE L | 90
	MOVE length
	ROTATE L | 90
END

FUN CUBE_CIRCLE | i | le
	CALL CUBE | le
	ROTATE L | 10
	VAL new | CALC i - 1
	CALL_IF new > 0 | CUBE_CIRCLE | new | le
END

FUN INC_CUBE_CIRCLE | steps
	CALL CUBE_CIRCLE | 6 | CALC steps * 1
	CALL CUBE_CIRCLE | 6 | CALC steps * 2
	CALL CUBE_CIRCLE | 6 | CALC steps * 2
	CALL CUBE_CIRCLE | 6 | CALC steps * 4
	CALL CUBE_CIRCLE | 6 | CALC steps * 5
	CALL CUBE_CIRCLE | 6 | CALC steps * 6
END

COLOR 255 | 0 | 0
CALL INC_CUBE_CIRCLE | 10
PEN_UP
MOVE 200
PEN_DOWN

COLOR 0 | 255 | 0
CALL INC_CUBE_CIRCLE | 10
ROTATE R | 90
PEN_UP
MOVE 200
PEN_DOWN

COLOR 0 | 0 | 255
CALL INC_CUBE_CIRCLE | 10
ROTATE R | 90
PEN_UP
MOVE 200
PEN_DOWN

COLOR 255 | 255 | 0
CALL INC_CUBE_CIRCLE | 20
